    ```typescript
    // D-08: soft fade-out tail when muting mid-cue. Time constant 0.05 → ~150 ms perceptual decay (3× constant).
    // Source: 03-RESEARCH.md Pattern 5 lines 471-481 + Pitfall 9 lines 575-579 fallback.
    const MUTE_FADE_TIME_CONSTANT = 0.05
    const MIN_GAIN_VALUE = 0.0001  // never ramp to 0 (exponentialRampToValueAtTime would throw)

    function applyMuteFadeOut(activeCue: CueHandle, audioCtx: AudioContext): void {
      const now = audioCtx.currentTime
      const gainParam = activeCue.envelope.gain
      // Modern browsers: cancelAndHoldAtTime is the right primitive. Safari <16.4: fall back.
      if (typeof gainParam.cancelAndHoldAtTime === 'function') {
        gainParam.cancelAndHoldAtTime(now)
      } else {
        // Fallback (Pitfall 9): freeze at current value, then ramp.
        const currentValue = gainParam.value
        gainParam.cancelScheduledValues(now)
        gainParam.setValueAtTime(currentValue, now)
      }
      gainParam.setTargetAtTime(MIN_GAIN_VALUE, now, MUTE_FADE_TIME_CONSTANT)
    }
    ```

    4. Define the factory:
    ```typescript
    const LEAD_IN_TICK_INTERVAL_SEC = 1.0
    const LEAD_IN_DURATION_SEC = 3.0
    const POST_CLOSE_DECAY_SEC = 0.2  // small grace after close so in-flight cue tail naturally rings out (D-16 + Pitfall 8)

    export async function createAudioEngine(): Promise<AudioEngine> {
      // D-09: AudioContext is created in this function, which is called synchronously from the
      // Start click handler in App.tsx. Browser autoplay policy MUST see a fresh user-gesture
      // chain or AC will start in 'suspended'.
      const audioCtx = new AudioContext()
      // If created suspended (Chrome occasionally does this), resume immediately.
      if (audioCtx.state === 'suspended') {
        await audioCtx.resume()
      }

      let activeCue: CueHandle | null = null
      let muted = false  // D-07: default false (audio ON)
      let closed = false
      let scheduler: SchedulerHandle | null = null  // reserved for future continuous-mode use; lead-in + boundary-driven scheduling do not require it.

      const engine: AudioEngine = {
        scheduleLeadIn(startAudioTime: number, _plan: BreathingPlan): number {
          if (closed) return startAudioTime + LEAD_IN_DURATION_SEC
          // 3 ticks at startAudioTime + 0, +1, +2 (D-14)
          // First In cue at startAudioTime + 3 (D-14: numerals replaced by first In phase label at t=0; gong fires)
          if (!muted) {
            scheduleTick(audioCtx, startAudioTime + 0, audioCtx.destination)
            scheduleTick(audioCtx, startAudioTime + 1, audioCtx.destination)
            scheduleTick(audioCtx, startAudioTime + 2, audioCtx.destination)
            activeCue = scheduleInCue(audioCtx, startAudioTime + LEAD_IN_DURATION_SEC, audioCtx.destination)
          }
          return startAudioTime + LEAD_IN_DURATION_SEC
        },
        scheduleNextCue({ newPhase, audioTime }: { newPhase: 'in' | 'out'; audioTime: number }): void {
          if (closed) return
          if (muted) return  // D-08 unmute-waits-for-boundary; if currently muted, skip this cue
          activeCue = newPhase === 'in'
            ? scheduleInCue(audioCtx, audioTime, audioCtx.destination)
            : scheduleOutCue(audioCtx, audioTime, audioCtx.destination)
        },
        setMuted(next: boolean): void {
          if (closed) {
            muted = next
            return
          }
          if (next && activeCue !== null) {
            // D-08: mute mid-cue → soft fade-out
            applyMuteFadeOut(activeCue, audioCtx)
          }
          // D-08: unmute mid-phase → do nothing (next cue plays at next boundary). NOT a make-up cue.
          muted = next
        },
        get muted(): boolean {
          return muted
        },
        now(): number {
          return audioCtx.currentTime
        },
        async close(): Promise<void> {
          if (closed) return
          closed = true
          if (scheduler !== null) {
            scheduler.stop()
            scheduler = null
          }
          // Pitfall 8: the in-flight cue tail will ring out naturally up to ~5 s; for D-16,
          // we close immediately and trust the audio thread's already-scheduled gain ramps to
          // complete. close() resolves when system audio resources are released.
          await audioCtx.close()
        },
      }

      return engine
    }
    ```

    ```typescript
          return engineRef.current.now() + 3
        }
        setStatus('starting')
        try {
          const engine = await createAudioEngine()
          engineRef.current = engine
          engine.setMuted(muted)
          // D-09 + D-13 + D-14: schedule the lead-in (3 ticks + first In cue at +3 s).
          // Anchor at engine.now() captured here — the audioCtx.currentTime instant of lead-in start.
          const startAudioTime = engine.now()
          const firstInCueTime = engine.scheduleLeadIn(startAudioTime, plan)
          setStatus('lead-in')
          setAudioAvailable(true)
          return firstInCueTime
        } catch {
          // D-10: visuals-only fallback. App.tsx still runs the visual countdown via setTimeout chain.
          setAudioAvailable(false)
          setStatus('failed')
          return null
        }
      }, [muted])

      const stop = useCallback(async (): Promise<void> => {
        const engine = engineRef.current
        if (engine !== null) {
          await engine.close()
          engineRef.current = null
        }
        setStatus('idle')
      }, [])

      const setMuted = useCallback((next: boolean): void => {
        setMutedState(next)
        engineRef.current?.setMuted(next)  // D-08 fade-out tail when muting mid-cue
      }, [])

      const notifyPhaseBoundary = useCallback(
        (args: { newPhase: 'in' | 'out'; audioTime: number }): void => {
          engineRef.current?.scheduleNextCue(args)
        },
        [],
      )

      const audioNow = useCallback((): number | null => {
        return engineRef.current?.now() ?? null
      }, [])

      return {
        status,
        audioAvailable,
        muted,
        start,
        stop,
        setMuted,
        notifyPhaseBoundary,
        audioNow,
      }
    }
    ```
