```typescript
// src/domain/sessionController.ts
export type SessionStatus = 'idle' | 'running' | 'complete';
export interface IdleSessionState { status: 'idle'; selectedSettings: SessionSettings; }
export interface RunningSessionState { status: 'running'; selectedSettings: SessionSettings; lockedSettings: SessionSettings; plan: BreathingPlan; startedAtMs: number; lastFrame: SessionFrame; }
export interface CompleteSessionState { status: 'complete'; selectedSettings: SessionSettings; lockedSettings: SessionSettings; plan: BreathingPlan; completedAtMs: number; message: 'Session complete'; }
export type SessionState = IdleSessionState | RunningSessionState | CompleteSessionState;
export function startSession(selectedSettings: SessionSettings, nowMs: number): RunningSessionState;
export function endSession(state: RunningSessionState | CompleteSessionState): IdleSessionState;
export function extendTimedSession(state: RunningSessionState, durationMinutes: number): RunningSessionState;
export function completeIfNeeded(state: RunningSessionState, nowMs: number): RunningSessionState | CompleteSessionState;
```
