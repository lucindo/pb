Mock SessionClock pattern (test-local — recommended fixture):
```
function createMockSessionClock(initialNow: number) {
  let nowValue = initialNow
  const suspendSubs = new Set<() => void>()
  const resumeSubs = new Set<() => void>()
  return {
    clock: {
      now: () => nowValue,
      schedule: () => {},
      setMasterGain: () => {},
      onSuspend: (cb) => { suspendSubs.add(cb); return () => suspendSubs.delete(cb) },
      onResume:  (cb) => { resumeSubs.add(cb);  return () => resumeSubs.delete(cb)  },
      onClose:   () => () => {},
    } satisfies SessionClock,
    advance: (delta: number) => { nowValue += delta },
    set: (value: number) => { nowValue = value },
    freeze: () => { /* no-op — advance/set are the only mutations */ },
    triggerSuspend: () => suspendSubs.forEach((cb) => cb()),
    triggerResume:  () => resumeSubs.forEach((cb) => cb()),
  }
}
```
